#include <iostream>
#include <vector>
#include <algorithm>
#include <string>
#include <queue>
using namespace std;
int main() {

	int a, b;
	cin >> a >> b;
	if (a > b) {
		int t = b;
		b = a;
		a = t;
	}
	if (a > 9 || b > 9 || a < 1 || b < 1) {
		cout << "INPUT ERROR!";
		return 0;
	}
	for (int j = 1; j < 10; j++) {
		for (int i = a; i <= b; i++) {
			cout << i << " * " << j << " = ";
			if (i * j < 10)
				cout << " " << i * j << "   ";
			else
				cout << i * j << "   ";
		}
		cout << endl;
	}

	return 0;
}
